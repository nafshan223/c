#include <stdio.h>
void main()
{
   int ams,x,rev=0,s;
   printf("enter a nmbr \n");
   scanf("%d",&ams);
   x=ams;
   while(x!=0)
   {
      s=x%10; 
      rev=rev*10+s;
      x=x/10;
     
   }
   if(rev == ams)
   {
       printf("%d is an palindrome",ams);
   }
   else
   {
       printf("%d is not an palindrome",ams);
   }
